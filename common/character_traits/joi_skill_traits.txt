REPLACE_OR_CREATE:basic_offensive_planner = {
	type = skill
	texture =gfx/interface/icons/character_trait_icons/basic_offensive_planner.dds

	command_modifier = {
		unit_offense_mult = 0.05
		character_popularity_add = 5
	}

	possible = {
		has_role_of_type = general
		owner ?= {
			has_technology_researched = napoleonic_warfare
		}
		NOR = {
			has_trait = experienced_offensive_planner
			has_trait = expert_offensive_planner
		}
		is_adult = yes
	}

	weight = {
		value = 50
		if = {
			limit = {
				has_trait = basic_defensive_strategist
			}
			add = -40
		}
	}

	replace = {
		inexperienced
	}

	value = 1
}
