			}

		ideology_bonapartist_valid_trigger = yes
	}

		NOT = { has_role_of_type = agitator }
		ideology_bonapartist_valid_trigger = yes
	}

	interest_group_leader_weight = {
		value = 200
		# less likely in a theocracy
		if = {
			limit = {
				owner = {
					has_law_or_variant = law_type:law_theocracy
				}
			}
			add = -75
		}
		# less likely if IG is happy in a republic
		if = {
			limit = {
				owner = {
					NOT = { has_law = law_type:law_monarchy }
				}
				ig_approval > 0
			}
			add = -100
		}

		# more likely for PB and AF

			limit = {
				OR = {
					is_interest_group_type = ig_petty_bourgeoisie
					is_interest_group_type = ig_armed_forces
				}
			}
			add = 50
		}

			limit = {
				owner = {
					has_law = law_type:law_monarchy
					NOT = { has_variable = bonapartist_restoration_var }
				}
				ig_approval < 0
			}
			add = 75
		}

		# reduced unhappy modifier if it's not a monarchy
		if = {
			limit = {
				owner = {
					NOT = { has_law = law_type:law_monarchy }
					NOT = { has_variable = bonapartist_restoration_var }
				}
				ig_approval < 0
			}
			add = 25
		}

		# conservative and/or reactionary leaders have a higher spawn chance if the Anti-Dreyfusards won.
		if = {
			limit = {
				has_variable = anti_dreyfusard_victory
			}
			add = 50
		}

		if = {
			limit = {
				OR = {
					has_variable = communist_ig_var
					has_variable = chose_vanguardism
					has_variable = chose_anarchism
					has_variable = chose_communism
					has_variable = chose_ethno_nationalism
					has_variable = chose_conservatism
					has_variable = chose_fascism
				}
			}
			multiply = 0.1
		}
	}

				}
			}
			add = -75
		}

			add = -100
		}

					is_interest_group_type = ig_armed_forces
				}
			}
			add = 50
		}

			limit = {
				OR = {
					has_role_of_type = general

				}
			}
			add = 50
		}
		# more likely if the IG is unhappy with the monarchy, and that monarchy is not Bonapartist

			limit = {
				owner = {
					has_law = law_type:law_monarchy
					NOT = { has_variable = bonapartist_restoration_var }
				}
				interest_group.ig_approval < 0
			}
			add = 75
		}

		# reduced unhappy modifier if it's not a monarchy
		if = {
			limit = {
				owner = {
					NOT = { has_law = law_type:law_monarchy }
					NOT = { has_variable = bonapartist_restoration_var }
				}
				interest_group.ig_approval < 0
			}
			add = 25
		}

		if = { # Legitimist agitators rare in Orleanist kingdoms
			limit = {
				has_role_of_type = agitator
				owner = {
					has_variable = bonapartist_restoration_var
