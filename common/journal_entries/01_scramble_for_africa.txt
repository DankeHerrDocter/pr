REPLACE_OR_CREATE:je_scramble_for_africa = {

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_foreign_affairs

	is_shown_when_inactive = {
		NOT = { is_country_type = unrecognized }
		has_technology_researched = quinine
		any_scope_state = {
			is_incorporated = no
			is_in_geographic_region = geographic_region_africa
		}
		NOT = { has_global_variable = berlin_conf_already_occured_before_var }
		OR = {
					country_rank >= rank_value:major_power
					AND = {
						exists = c:GBR
						THIS = c:GBR
					}
					AND = {
						exists = c:FRA
						THIS = c:FRA
					}
					AND = {
						exists = c:GER
						THIS = c:GER
					}
					AND = {
						exists = c:BEL
						THIS = c:BEL
					}
					AND = {
						exists = c:POR
						THIS = c:POR
					}
					AND = {
						exists = c:SPA
						THIS = c:SPA
					}
					AND = {
						exists = c:GER
						THIS = c:GER
					}
				}
	}

	possible = {
		OR = {
			has_technology_researched = civilizing_mission
			custom_tooltip = {
			text = scramble_for_africa_has_start
			has_global_variable = scramble_for_africa_has_start_var
			}
		}
		country_rank >= rank_value:major_power		
		OR = {
			any_scope_state = {
				is_in_geographic_region = geographic_region_africa
				is_incorporated = no
				is_treaty_port = no
			}
			any_subject_or_below = {
				is_in_geographic_region = geographic_region_africa
			}
		}
	}

	immediate = {
		trigger_event = {
			id = africa_colonial_events.1
			popup = yes
		}
		save_scope_as = scramble_for_africa_self

		set_variable = {
			name = controlled_colonies_in_africa_cached
			value = controlled_colonies_in_africa
		}
		set_global_variable = { name = scramble_for_africa_global_var months = 6 }
		if = {
			limit = {
				NOT = { has_global_variable = scramble_for_africa_has_start_var }
			}
			set_global_variable = scramble_for_africa_has_start_var
		}
		set_variable = berlin_conference_begin
		if = {#设定固定国家变量
			limit = {
				exists = c:FRA
				THIS = c:FRA 
			}
			set_variable = berlin_conference_begin_fra
			}
		if = {
			limit = {
				exists = c:GBR
				THIS = c:GBR 
			}
			set_variable = berlin_conference_begin_gbr
			}
		if = {
			limit = {
				exists = c:POR
				THIS = c:POR 
			}
			set_variable = berlin_conference_begin_por
			}
		if = {
			limit = {
				exists = c:BEL
				THIS = c:BEL 
			}
			set_variable = berlin_conference_begin_bel
			}
		if = {
			limit = {
				exists = c:SPA
				THIS = c:SPA 
			}
			set_variable = berlin_conference_begin_spa
			}
		if = {
			limit = {
				exists = c:GER
				THIS = c:GER 
			}
			set_variable = berlin_conference_begin_ger
		}
	}

	complete = {
		custom_tooltip = {
			text = scramble_for_africa_complete_2_tt
			any_state_region_in_africa = {
				any_scope_state = {
					owner = {
						OR = {
							is_country_type = recognized
							top_overlord ?= {
								is_country_type = recognized
							}
						}
					}
				}
				percent >= 0.95
			}
		}
		NOT = { has_global_variable = scramble_for_africa_global_var }
	}

	on_complete = {
		set_global_variable = berlin_conf_already_occured_before_var
		if = {
			limit = {
				has_game_rule = berlin_conference_enable_event
			}
			custom_tooltip = {
			text = berlin_conference_begin_1_tt
			if = {
			limit = {
				NOT = { has_variable = berlin_conference_begin }
			}
			set_variable = berlin_conference_begin
			}
		if = {#设定固定国家变量
			limit = {
				exists = c:FRA
				THIS = c:FRA 
				NOT = { has_variable = berlin_conference_begin_fra }
			}
			set_variable = berlin_conference_begin_fra
			}
		if = {
			limit = {
				exists = c:GBR
				THIS = c:GBR 
				NOT = { has_variable = berlin_conference_begin_gbr }
			}
			set_variable = berlin_conference_begin_gbr
			}
		if = {
			limit = {
				exists = c:POR
				THIS = c:POR 
				NOT = { has_variable = berlin_conference_begin_por }
			}
			set_variable = berlin_conference_begin_por
			}
		if = {
			limit = {
				exists = c:BEL
				THIS = c:BEL 
				NOT = { has_variable = berlin_conference_begin_bel }
			}
			set_variable = berlin_conference_begin_bel
			}
		if = {
			limit = {
				exists = c:SPA
				THIS = c:SPA 
				NOT = { has_variable = berlin_conference_begin_spa }
			}
			set_variable = berlin_conference_begin_spa
			}
		if = {
			limit = {
				exists = c:GER
				THIS = c:GER 
				NOT = { has_variable = berlin_conference_begin_ger }
			}
			set_variable = berlin_conference_begin_ger
			}
		}
		trigger_event = { id = joi_flavor_colony.1 }#柏林会议
		}
		if = {
			limit = {
				has_game_rule = berlin_conference_disable_event
			}
			trigger_event = { id = africa_colonial_events.2 }
		}
		if = {
			limit = {
				has_variable = controlled_colonies_in_africa_cached
			}
			remove_variable = controlled_colonies_in_africa_cached
		}
	}

	fail = {
		custom_tooltip = {
			text = scramble_for_africa_fail_tt
			NOR = {
				any_scope_state = {
					is_incorporated = no
					is_in_geographic_region = geographic_region_africa
				}
				any_subject_or_below = {
					is_in_geographic_region = geographic_region_africa
				}
			}
		}
	}

	on_fail = {
		#trigger_event = {
		#	id = africa_colonial_events.3
		#	days = 0
		#	popup = yes
		#}
		if = {
			limit = {
				has_variable = controlled_colonies_in_africa_cached
			}
			remove_variable = controlled_colonies_in_africa_cached
		}
	}

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = scramble_for_africa_6_states
				trigger = {
					any_scope_state = {
						count >= 6
						is_incorporated = no
						is_in_geographic_region = geographic_region_africa
					}
				}
			}
			triggered_desc = {
				desc = scramble_for_africa_4_states
				trigger = {
					any_scope_state = {
						count >= 4
						is_incorporated = no
						is_in_geographic_region = geographic_region_africa
					}
				}
			}
			triggered_desc = {
				desc = scramble_for_africa_3_states
				trigger = {
					any_scope_state = {
						count >= 1
						is_incorporated = no
						is_in_geographic_region = geographic_region_africa
					}
				}
			}
			triggered_desc = {
				desc = scramble_for_africa_0_states
			}
		}
	}

	# No timeout. Imperialism never ends.
	# timeout = 7300

	on_timeout = {

	}

	on_weekly_pulse = {
		effect = {
			set_variable = {
				name = controlled_colonies_in_africa_cached
				value = controlled_colonies_in_africa
			}
		}
	}

	on_yearly_pulse = {
		random_events = {
			100 = 0
			10 = africa_colonial_events.4
			10 = africa_colonial_events.5
			10 = africa_colonial_events.6
			5 = africa_colonial_events.7
			10 = patagonia.2
			10 = amazon.2

			10 = africa_colonial_events.40
			10 = africa_colonial_events.41
			10 = africa_colonial_events.51
			10 = africa_colonial_events.53
			10 = africa_colonial_events.54
		}
	}

	weight = 200
	should_be_pinned_by_default_uninvolved_or_context = yes
}
