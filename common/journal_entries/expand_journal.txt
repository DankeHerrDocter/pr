establish_national_academy = {#建立国立学术院
	icon = "gfx/interface/icons/invention_icons/academia.dds"

	group = je_group_internal_affairs

	is_shown_when_inactive = {		
		country_rank >= rank_value:major_power
		NOT = { has_variable = this_country_has_national_academy }
		has_technology_researched = dialectics
	}

	possible = {
		has_technology_researched = dialectics
		has_technology_researched = napoleonic_warfare
		has_technology_researched = railways
		ig:ig_intelligentsia = {
			is_in_government = yes
		}
	}

	immediate = {
		set_variable = { name = establish_national_academy_var value = 0 }
		capital = {
			save_scope_as = national_academy_state
		}
	}

	scripted_button = start_national_academy_project_button

	on_monthly_pulse = {
		effect = {
			if = {
				limit = { has_modifier = start_national_academy_project_modifier }
				change_variable = { name = establish_national_academy_var add = 1 }
			}
		}
	}

	complete = {
		custom_tooltip = {
		text = establish_national_academy_complete_tt
		scope:journal_entry = {
			is_goal_complete = yes
			}
		}
	}

	on_complete = {
		set_variable = this_country_has_national_academy
		if = {
			limit = { has_modifier = start_national_academy_project_modifier }
			remove_modifier = start_national_academy_project_modifier
		}
		add_loyalists = { value = 0.05 interest_group = ig:ig_intelligentsia }
		if = {
			limit = { c:USA ?= THIS }
			random_scope_state = {
				limit = { is_capital = yes }
				add_modifier = us_academy_of_sciences_modifier
			}
		}
		if = {
			limit = { c:CHI ?= THIS }
			random_scope_state = {
				limit = { is_capital = yes }
				add_modifier = china_academia_sinica_modifier
			}
		}
		if = {
			limit = { c:CAN ?= THIS }
			random_scope_state = {
				limit = { is_capital = yes }
				add_modifier = canada_national_research_council_modifier
			}
		}
		if = {
			limit = { c:JAP ?= THIS }
			random_scope_state = {
				limit = { is_capital = yes }
				add_modifier = jap_imperial_academy_modifier
			}
		}
		if = {
			limit = { c:AST ?= THIS }
			random_scope_state = {
				limit = { is_capital = yes }
				add_modifier = australian_academy_of_science_modifier
			}
		}
		if = {
			limit = { c:BIC ?= THIS }
			random_scope_state = {
				limit = { is_capital = yes }
				add_modifier = indian_national_science_academy_modifier
			}
		}
		if = {
			limit = { c:BHT ?= THIS }
			random_scope_state = {
				limit = { is_capital = yes }
				add_modifier = indian_national_science_academy_modifier
			}
		}
		if = {
			limit = { c:KOR ?= THIS }
			random_scope_state = {
				limit = { is_capital = yes }
				add_modifier = korean_national_science_academy_modifier
			}
		}
		if = {
			limit = { c:AUS ?= THIS }
			random_scope_state = {
				limit = { is_capital = yes }
				add_modifier = osterreichische_akademie_der_wissenschaften_modifier
			}
		}
		if = {
			limit = {
				NOR = {
					c:USA ?= THIS
					c:CHI ?= THIS
					c:CAN ?= THIS
					c:JAP ?= THIS
					c:AST ?= THIS
					c:BIC ?= THIS
					c:BHT ?= THIS
					c:KOR ?= THIS
					c:AUS ?= THIS
				}
			}
			random_scope_state = {
				limit = { is_capital = yes }
				add_modifier = national_academy_modifier
			}
		}
	}

	fail = {
		in_default = yes
	}

	on_fail = {
		if = {
			limit = { has_modifier = start_national_academy_project_modifier }
			remove_modifier = start_national_academy_project_modifier
		}
		add_radicals = { value = 0.1 interest_group = ig:ig_intelligentsia }
	}

	current_value = {
		value = var:establish_national_academy_var
	}

	goal_add_value = {
		value = 60
	}

	progressbar = yes

	weight = 1000

	should_be_pinned_by_default_uninvolved_or_context = yes
}
