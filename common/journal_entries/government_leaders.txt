government_leader_nicholas_i = {#尼古拉一世：东正教，独裁制，民族性
	icon = "gfx/interface/icons/joi_icons/government_leader_nicholas_i.png"

	group = je_group_government_leaders

	is_shown_when_inactive = {
		country_has_primary_culture = cu:russian
		any_scope_character = {
			has_template = RUS_nicholas_i
		}
		has_law_or_variant = law_type:law_monarchy
	}

	possible = {
		any_scope_character = {
			custom_tooltip = {
			text = government_leader_nicholas_i_possible_tt
			has_template = RUS_nicholas_i
			}
			is_ruler_of_own_country = yes
		}
		has_law_or_variant = law_type:law_monarchy
		has_law_or_variant = law_type:law_autocracy
		has_law_or_variant = law_type:law_state_religion
	}

	immediate = {
		set_variable = government_leader_government_leader_nicholas_i_var
		set_variable = { name = government_leader_nicholas_i_counter value = 0 }
		random_character = {
			limit = { has_template = RUS_nicholas_i }
			save_scope_as = nicholas_i_scope
		}
	}

	modifiers_while_active = {
		government_leader_nicholas_i_je_modifier
	}

	complete = {
		OR = {
			custom_tooltip = {
			text = government_leader_nicholas_i_complete_tt
			NOT = {
				ruler ?= { has_template = RUS_nicholas_i }
				}
			}
			scope:nicholas_i_scope ?= {
				is_ruler_of_own_country = yes
				age >= 58
			}
		}
	}

	on_complete = {
		custom_tooltip = {
		text = government_leader_nicholas_i_on_complete_tt
		government_leader_nicholas_i_effect = yes
		}
		trigger_event = { id = joi_flavor_gov.1 }
	}

	fail = {
		OR = {
			NOT = { has_law_or_variant = law_type:law_monarchy }
			NOT = { has_law_or_variant = law_type:law_autocracy }
			NOT = { has_law_or_variant = law_type:law_state_religion }
		}
	}

	weight = 100

	should_be_pinned_by_default_uninvolved_or_context = yes
}

government_leader_william_gladstone = {#威廉·尤尔特·格莱斯顿：格莱斯顿式自由
	icon = "gfx/interface/icons/joi_icons/william_gladstone.png"

	group = je_group_government_leaders

	is_shown_when_inactive = {
		country_has_primary_culture = cu:british
		OR = {
			any_scope_character = {
				has_template = gbr_gladstone_character_template
			}
			ig:ig_intelligentsia = {
				is_in_government = yes
			}
		}
		country_has_voting_franchise = yes
	}

	possible = {
		any_scope_character = {
			custom_tooltip = {
			text = government_leader_william_gladstone_possible_tt
			has_template = gbr_gladstone_character_template
			}
			has_role_of_type = politician
			prominence >= 50
		}
		ig:ig_intelligentsia = { is_in_government = yes }
		OR = {
			has_law_or_variant = law_type:law_wealth_voting
			has_law_or_variant = law_type:law_census_voting
			has_law_or_variant = law_type:law_universal_suffrage
		}
	}

	immediate = {
		ig:ig_intelligentsia = {
        	set_variable = { name = memento_mori_immunity years = 30 }
        	save_scope_as = intelligentsia_scope
        }
		random_character = {
			limit = { has_template = gbr_gladstone_character_template }
			save_scope_as = gladstone_scope
			set_as_interest_group_leader = yes
		}
		random_active_party = {
			limit = {
				has_party_member = scope:intelligentsia_scope
			}
			save_scope_as = liberal_party
		}
		set_variable = { name = government_leader_william_gladstone_counter value = 0 }
	}

	modifiers_while_active = {
		government_leader_william_gladstone_je_modifier
	}

	complete = {
		OR = {
			custom_tooltip = {
			text = government_leader_william_gladstone_complete_tt
			NOT = {
				any_scope_character = {
					has_template = gbr_gladstone_character_template
					}
				}
			}
			scope:gladstone_scope ?= {
				has_role_of_type = politician
				age >= 75
			}
		}
	}

	on_complete = {
		custom_tooltip = {
		text = government_leader_william_gladstone_on_complete_tt
		government_leader_william_gladstone_effect = yes
		}
		trigger_event = { id = joi_flavor_gov.2 }
		custom_tooltip = {
		text = government_leader_william_gladstone_on_complete_2_tt
		}
		show_as_tooltip = {
			active_law:lawgroup_trade_policy ?= { # 格莱斯顿式自由主义
				add_amendment = {
					type = amendment_gladstone_liberalism_dogmas
					sponsor = PREV.ig:ig_intelligentsia
				}
			}
		}
	}

	fail = {
		OR = {
			custom_tooltip = {
			text = country_no_voting_franchise_tt
			country_has_voting_franchise = no
			}
			ig:ig_intelligentsia = {
				is_marginal = yes
			}
		}
	}

	on_fail = {
		set_variable = government_leader_william_gladstone_fail_var
		trigger_event ={ id = joi_flavor_gov.2 }
	}

	weight = 100

	should_be_pinned_by_default_uninvolved_or_context = yes
}
