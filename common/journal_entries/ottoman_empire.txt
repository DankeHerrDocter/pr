	group = je_group_movement_for_democracy

	is_shown_when_inactive = {
		c:TUR ?= THIS
		has_law_or_variant = law_type:law_census_voting
		has_law_or_variant = law_type:law_universal_suffrage
	}

	possible = {
		OR = {
			AND = {
				custom_tooltip = {
				text = liberalism_of_turkey_possible_1_tt
				has_variable = ottoman_constitutional_era_complete
				}
				custom_tooltip = {
				text = liberalism_of_turkey_possible_2_tt
				has_global_variable = great_ottomanism_finish_var
				}
			}
			has_law_or_variant = law_type:law_presidential_republic
			has_law_or_variant = law_type:law_parliamentary_republic
		}
	}

	immediate = {
		set_variable = { name = liberalism_of_turkey_var value = 0 }
		ruler = {
			set_variable = liberalism_leader_var
			save_scope_as = liberalism_leader
			set_ideology = ideology:ideology_kemalism_leader
		}
	}

	modifiers_while_active = {
		liberalism_of_turkey_modifier
	}

	complete = {
		custom_tooltip = {
		text = liberalism_of_turkey_complete_2_tt
		has_law_or_variant = law_type:law_universal_suffrage
		has_law_or_variant = law_type:law_womens_suffrage
		has_law_or_variant = law_type:law_presidential_republic
		has_law_or_variant = law_type:law_parliamentary_republic
		has_law_or_variant = law_type:law_right_to_associate
		has_law_or_variant = law_type:law_public_schools
		has_law_or_variant = law_type:law_protected_speech
		has_law_or_variant = law_type:law_interventionism
		has_law_or_variant = law_type:law_total_separation
		ig:ig_devout = {
			is_in_government = no
			is_powerful = no
		}
		any_political_movement = {
			OR = {
				is_political_movement_type = movement_cultural_majority
				is_political_movement_type = movement_cultural_minority
			}
			political_movement_radicalism < 0.5
		}
		any_political_movement = {
			OR = {
				is_political_movement_type = movement_religious_majority
				is_political_movement_type = movement_religious_minority
			}
			political_movement_radicalism < 0.25
			}
		}
	}

	on_complete = {
		liberalism_of_turkey_variable_effect = yes
		set_variable = liberalism_of_turkey_completed
		trigger_event = { id = joi_flavor_tur.101 }
	}

	fail = {
		custom_tooltip = {
		text = liberalism_of_turkey_complete_tt
		scope:liberalism_leader ?= { is_ruler_of_own_country = no }
		}
	}

	on_fail = {
		liberalism_of_turkey_variable_effect = yes
		trigger_event = { id = joi_flavor_tur.101 }
	}

	invalid = {
		NOT = {
			OR = {
				has_law_or_variant = law_type:law_universal_suffrage
				has_law_or_variant = law_type:law_census_voting
			}
		}
	}

	status_desc = {
		first_valid = {
			triggered_desc = { 
				desc = liberalism_of_turkey_1
				trigger = {
					has_law_or_variant = law_type:law_universal_suffrage
					has_law_or_variant = law_type:law_womens_suffrage
					has_law_or_variant = law_type:law_public_schools
					has_law_or_variant = law_type:law_protected_speech
					has_law_or_variant = law_type:law_total_separation
					has_law_or_variant = law_type:law_right_to_associate
					ig:ig_devout = {
						is_in_government = no
						is_powerful = no
					}
				}
			}
			triggered_desc = { 
				desc = liberalism_of_turkey_2
				trigger = {
					has_law_or_variant = law_type:law_universal_suffrage
					has_law_or_variant = law_type:law_womens_suffrage
					has_law_or_variant = law_type:law_public_schools
					has_law_or_variant = law_type:law_protected_speech
					has_law_or_variant = law_type:law_total_separation
				}
			}
			triggered_desc = { 
				desc = liberalism_of_turkey_3
				trigger = {
					OR = {
						has_law_or_variant = law_type:law_womens_suffrage
						has_law_or_variant = law_type:law_total_separation
					}
				}
			}
			triggered_desc = { 
				desc = liberalism_of_turkey_4
				trigger = {
					NOT = {
						OR = {
							has_law_or_variant = law_type:law_womens_suffrage
							has_law_or_variant = law_type:law_total_separation
						}
					}
				}
			}
		}
	}

	weight = 100

	should_be_pinned_by_default_uninvolved_or_context = yes
}
