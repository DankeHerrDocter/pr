			custom_tooltip = {

					is_goal_complete = yes
					}
				}
			custom_tooltip = {
				text = conservative_order_prussia_constitution_complete_2_tt

			}

		change_variable = { name = conservative_order_prussia_var add = 1 }
		}
	}

	fail = {
		NOT = { has_journal_entry = conservative_order_prussia }

	current_value = {
