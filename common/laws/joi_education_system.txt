law_madrasa = {#伊斯兰学校
	group = lawgroup_education_system
	parent = law_religious_schools 

	icon = "gfx/interface/icons/law_icons/religious_schools.dds"

	progressiveness = -50

	requires_law_or = {
		law_state_religion
		law_millet_system
		law_people_of_the_book
	}

	is_visible = {
		OR = {
			country_has_state_religion = rel:sunni
			country_has_state_religion = rel:shiite
			country_has_state_religion = rel:ibadi
		}
	}

	can_enact = {
		OR = {
			country_has_state_religion = rel:sunni
			country_has_state_religion = rel:shiite
			country_has_state_religion = rel:ibadi
		}
	}

	on_enact = {
		if = {
			limit = {
				ig:ig_landowners = {
					if = {
						limit = {
							NOT = { has_ideology = ideology:ideology_pious }
						}
						add_ideology = ideology_pious
					}
				}
			}
		}
	}

	modifier = {
		country_institution_cost_institution_schools_mult = -0.5
		building_university_throughput_add = -0.2
		country_loyalism_increases_full_acceptance_mult = 0.1
	}

	institution = institution_schools
	institution_modifier = {
		state_conversion_mult = 0.5
		state_education_access_wealth_add = 0.0025
		interest_group_ig_devout_pol_str_mult = 0.3
		country_weekly_innovation_max_add = -5
	}

	ai_will_do = {
		always = no
	}

	ai_enact_weight_modifier = { #Petitions
		value = 50
	}
}
