# Special isolationism variant for Japan
law_kyushu_system = {
	group = lawgroup_trade_policy
	parent = law_isolationism # Inherits ideological stances from parent

	icon = "gfx/interface/icons/law_icons/isolationism.dds"

	progressiveness = -50

	requires_law_or = {
		law_traditionalism
	}

	is_visible = {
		has_law = law_type:law_kyushu_system
		country_has_primary_culture = cu:japanese
		any_scope_state = {
			state_region = s:STATE_KYUSHU
			is_largest_state_in_region = yes
		}
	}

	can_enact = {
		is_in_power_bloc = no
		country_has_primary_culture = cu:japanese
		any_scope_state = {
			state_region = s:STATE_KYUSHU
			is_largest_state_in_region = yes
		}
	}

	modifier = {
		country_authority_mult = 0.25
		state_tax_capacity_mult = 0.15
		country_tech_spread_mult = -0.10
		state_trade_capacity_mult = -0.5
		state_tariff_import_add = 0.50
		state_tariff_export_add = 0.50
		state_subvention_export_add = 0.20
		state_subvention_import_add = 0.20

		country_leverage_resistance_mult = 0.75

		country_max_declared_interests_mult = -0.33

		country_disallow_trade_outside_kyushu_bool = yes
	}

	ai_will_do = {
		OR = {
			has_journal_entry = je_gyojo_shinwon
			ruler = {
				OR = {
					has_ideology = ideology:ideology_traditionalist
					has_ideology = ideology:ideology_vanguardist
				}
			}
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
}

law_imperial_preference = {	#帝国特惠制
	group = lawgroup_trade_policy
	parent = law_protectionism 

	icon = "gfx/interface/icons/law_icons/imperial_preference.dds"

	progressiveness = 0

	is_visible = {
		OR = {
			c:GBR ?= THIS
			c:IMP ?= THIS
			trigger_if = {
				limit = { exists = c:GBR }
				is_subject_of = c:GBR
			}
			trigger_if = {
				limit = { exists = c:IMP }
				is_subject_of = c:IMP
			}
		}
	}

	can_enact = {
		custom_tooltip = {
		text = imperial_preference_enable_tt
		has_variable = imperial_preference_complete_var
		}
	}

	modifier = {
		state_tariff_import_add = 0.75
		state_tariff_export_add = 0.50
		state_subvention_export_add = 0.60
		state_subvention_import_add = 0.40
		country_leverage_resistance_mult = 0.5
		state_export_advantage_mult = 0.15
	}

	ai_will_do = {
		always = yes
	}

	ai_enact_weight_modifier = { #Petitions
		value = 100
	}
}
