on_game_started = {
	effect = {
		# Memento Mori
		every_country = {
			every_interest_group = {
				leader = {
					if = {
						limit = {
							NOT = {
								has_variable = memento_mori_immunity
							}
						}
						set_variable = {
							name = memento_mori_immunity
							years = 5
						}
					}
				}
			}
		}
		set_global_variable = { name = ger_bel_score value = 0 }
        set_global_variable = {name = gbr_bel_score value = 0 }
        set_global_variable = { name = fra_bel_score value = 0 }
		c:BRZ ?= {
			add_modifier = {
				name = guerra_dos_farrapos_modifier
				years = 2
			}
		}
		random_character = {
			limit = {
				has_template = gbr_wellington_character_template
			}
			set_character_immortal = yes
		}
		every_country = {
			limit = {
				any_primary_culture = {
					OR = { 
						cu:south_german = this 
						cu:north_german = this
					}
				}
				NOT = { c:UBD ?= THIS }
				NOT = { c:TRS ?= THIS }
				NOT = {
					any_active_law = { has_amendment = amendment_type:amendment_landwehr_system }
				}
			}
			active_law:lawgroup_army_model ?= {
				add_amendment = {
					type = amendment_landwehr_system
					sponsor = PREV.ig:ig_armed_forces
				}
			}
		}
		every_country = {
			limit = {
				country_rank >= rank_value:unrecognized_regional_power
			}
			joi_set_amendment_var_efffect = yes
		}
		every_state = {
			limit = {
				region = sr:region_indonesia
				owner = c:DEI
			}
			add_modifier = cultivation_system_state_modifier
		}
		create_gbr_aus_start_military_army_general_effect = yes
		c:NET ?= {
			add_company = company_type:company_admiralty_rijkswerf
			company:company_admiralty_rijkswerf = {
				set_company_establishment_date = 1655.1.1
				set_company_state_region = s:STATE_HOLLAND
				}
			}
		s:STATE_HOLLAND = {
                region_state:NET = {
                    create_building = {
                        building = "building_shipyard"
                        add_ownership = {
                            company = {
                                type = company_admiralty_rijkswerf
                                country = "c:NET"
                                levels = 5
                            }
                        }
                	reserves = 1
           		}
        	}
		}
	}
}

joi_gbr_monarchy_action = {
	effect = {
		if = {
			limit = {
				OR = {
					has_law_or_variant = law_type:law_monarchy
					has_law_or_variant = law_type:law_social_monarchy
				}
				OR = {
					c:GBR ?= this
					c:IMP ?= this
				}
				game_date >= 1841.11.09
				ruler ?= {
					has_template = gbr_queen_victoria_template
				}
				has_variable = house_of_romanov_hannover
			}
			create_character = {
				template = gbr_rus_alex_edward_template
				heir = yes
			}
			set_variable = edward_vii_heir_var
		}
		if = {
			limit = {
				OR = {
					has_law_or_variant = law_type:law_monarchy
					has_law_or_variant = law_type:law_social_monarchy
				}
				OR = {
					c:GBR ?= this
					c:IMP ?= this
				}
				game_date >= 1865.06.03
				ruler ?= {
					has_template = gbr_rus_alex_edward_template
				}
				has_variable = house_of_romanov_hannover
			}
			create_character = {
				template = gbr_rus_george_v_template
				heir = yes
			}
			set_variable = george_v_heir_var
		}
		if = {
			limit = {
				OR = {
					has_law_or_variant = law_type:law_monarchy
					has_law_or_variant = law_type:law_social_monarchy
				}
				OR = {
					c:GBR ?= this
					c:IMP ?= this
				}
				game_date >= 1910.5.06
				ruler ?= {
					has_template = gbr_rus_alex_edward_template
				}
				heir = {
					has_template = gbr_rus_george_v_template
				}
				NOT = {
				has_variable = george_v_ruler_var
				}
			}
			trigger_event = { id = britain_monarchy.1 popup = yes }
		}
	}
}

joi_reset_aferican_brith_rate_action = {
	effect = {
		if = {
		limit = {
			OR = {
				region = sr:region_north_africa
				region = sr:region_nile_basin
				region = sr:region_west_africa
				region = sr:region_equatorial_africa
				region = sr:region_east_africa
				region = sr:region_southern_africa
			}
			NOT = { has_modifier = african_population_growth_modifier }
		}
		add_modifier = african_population_growth_modifier
		}
	}
}
