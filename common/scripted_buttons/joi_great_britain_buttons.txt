british_empire_india_button = {
	name = "british_empire_india_button"
	desc = "british_empire_india_button_desc"

	visible = {
		always = yes
	}

	cooldown = { months = 3 }

	possible = {
		bureaucracy >= 5000
		custom_tooltip = {
			text = british_empire_india_button_tt
			NOT = { 
				has_variable = annex_indian_mission
			}
		}
		OR = {
			AND = {
				any_subject_or_below = {
				is_player = no
				has_variable = princely_state
				}
			}
			AND = {
				c:BIC = {
					any_subject_or_below = {
					is_player = no
					has_variable = princely_state
					}
				}
			}
		}
	}

	ai_chance = {
		value = 10
	}

	effect = {
		set_variable = {
			name = annex_indian_mission
			months = 3
		}
		if = {
			limit = {
				NOT = { exists = c:BIC }
					any_subject_or_below = {
					is_player = no
					has_variable = princely_state
				}
			}
			add_radicals = { value = 0.2 religion = rel:hindu }
			add_radicals = { value = 0.1 religion = rel:sunni }
			custom_tooltip = {
			text = bic_annex_princely_states_tt
				every_state = {
					limit = {
						OR = {
							region = sr:region_south_india
							state_region = s:STATE_GUJARAT
							state_region = s:STATE_SINDH
							state_region = s:STATE_BALUCHISTAN
							state_region = s:STATE_MALWA
							state_region = s:STATE_RAJPUTANA
						}
					}
					add_modifier = princely_state_privilege_modifier
				}
				every_state = {
					limit = {
						owner = {
							is_subject_of = ROOT
							is_player = no
							has_variable = princely_state
							}
						}
					set_state_owner = ROOT
					add_modifier = princely_state_privilege_modifier
				}
			}
		}
		else_if = {
			limit = {
				exists = c:BIC
				c:BIC = {
					any_subject_or_below = {
					is_player = no
					has_variable = princely_state
					}
				}
			}
			c:BIC ?= {
				add_radicals = { value = 0.2 religion = rel:hindu }
				add_radicals = { value = 0.1 religion = rel:sunni }
			}
			custom_tooltip = {
			text = bic_annex_princely_states_tt
				every_state = {
					limit = {
						OR = {
							region = sr:region_south_india
							state_region = s:STATE_GUJARAT
							state_region = s:STATE_SINDH
							state_region = s:STATE_BALUCHISTAN
							state_region = s:STATE_MALWA
							state_region = s:STATE_RAJPUTANA
						}
					}
					add_modifier = princely_state_privilege_modifier
				}
				every_state = {
					limit = {
						owner = {
							is_subject_of = c:BIC
							is_player = no
							has_variable = princely_state
							}
						}
					set_state_owner = c:BIC
				}
			}
		}
	}
}
