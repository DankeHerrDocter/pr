	desc = "general_staff_1_button_desc"

	visible = {
		always = yes
	}

	possible = {
		custom_tooltip = {
			text = general_staff_1_button_tt
			NOT = {

		}

			name = has_general_staff_1
			years = 5
		}
		add_modifier = {
			name = staff_officer

		}
	}
}

conservative_order_prussia_constitution_button = {#在选举权上做出妥协
	name = "conservative_order_prussia_constitution_button"
	desc = "conservative_order_prussia_constitution_button_desc"

	visible = {
		always = yes
	}

	possible = {
		OR = {
			has_law_or_variant = law_type:law_autocracy
			has_law_or_variant = law_type:law_oligarchy
			has_law_or_variant = law_type:law_landed_voting
		}
		custom_tooltip = {
		text = buttton_not_been_used_tt
		NOT = { has_variable = conservative_order_prussia_constitution_button_var }	
		}
		OR = {
			any_civil_war = {
				is_civil_war_type = revolution
				civil_war_progress >= 0.25
			}
			ig:ig_intelligentsia ?= {
				is_powerful = yes
			}
		}
	}

	ai_chance = {
		value = 100
	}

	effect = {
		custom_tooltip = {
		text = conservative_order_prussia_constitution_button_tt
		set_variable = conservative_order_prussia_constitution_button_var 
		}
		if = {
			limit = { country_has_voting_franchise = no }
			activate_law = law_type:law_wealth_voting
		}
		else_if = {
			limit = { country_has_voting_franchise = yes }
			activate_law = law_type:law_census_voting
		}
		add_loyalists = { value = 0.05 }
		add_radicals = { value = 0.2 interest_group = ig:ig_landowners }
	}
}
