	}

		limit = {
			has_law_or_variant = law_type:law_secret_police
		}
		change_variable = { name = government_leader_nicholas_i_counter add = 1 }
	}
	if = {
		limit = {
			has_law_or_variant = law_type:law_censorship
			has_law_or_variant = law_type:law_outlawed_dissent
		}
		change_variable = { name = government_leader_nicholas_i_counter add = 1 }
	}
	if = {
		limit = {
			has_law_or_variant = law_type:law_secret_police
		}
		change_variable = { name = government_leader_nicholas_i_counter add = 1 }
	}
	if = {
		limit = {
			literacy_rate < 0.15
			bureaucracy >= 1000 
		}
		change_variable = { name = government_leader_nicholas_i_counter add = 1 }
	}
	if = {
		limit = {
			OR = {
				any_political_movement = {
					OR = {
						movement_is_liberal = yes
						movement_is_socialist = yes
					}
					political_movement_radicalism < 0.1
				}
				any_political_movement = {
					OR = {
						movement_is_liberal = yes
						movement_is_socialist = yes
					}

				}
			}
		}
		change_variable = { name = government_leader_nicholas_i_counter add = 1 }
	}
	if = {
		limit = {
			ig:ig_devout ?= {
				is_powerful = yes
				is_in_government = yes
			}
			ig:ig_landowners ?= {
				is_powerful = yes
				is_in_government = yes
			}
		}
		change_variable = { name = government_leader_nicholas_i_counter add = 1 }
	}
	if = {
		limit = {
			NOT = {
				any_country = {
				capital = {
                    is_in_geographic_region = geographic_region_europe
                }
                country_rank >= rank_value:major_power
                NOT = {
                	has_law_or_variant = law_type:law_monarchy
                	has_law_or_variant = law_type:law_social_monarchy

                	}
				}
			}
		}
		change_variable = { name = government_leader_nicholas_i_counter add = 1 }
	}
}

government_leader_william_gladstone_effect = {
	if = {
		limit = {
			has_law_or_variant = law_type:law_laissez_faire
			has_global_highest_gdp = yes
		}
		change_variable = { name = government_leader_william_gladstone_counter add = 1 }
	}
	if = {
		limit = {
			has_law_or_variant = law_type:law_free_trade
			market_capital.market = {
				market_trade = {
				  value >= 500000
				}
			}
		}
		change_variable = { name = government_leader_william_gladstone_counter add = 1 }
	}
	if = {
		limit = {
			tax_level <= low
			num_taxed_goods <= 5
		}
		change_variable = { name = government_leader_william_gladstone_counter add = 1 }
	}
	if = {
		limit = {
			investment_pool_gross_income >= 3000000
			has_healthy_economy = yes
		}
		change_variable = { name = government_leader_william_gladstone_counter add = 1 }
	}
	if = {
		limit = {
			has_variable = elementary_education_1870_act_var
			literacy_rate >= 0.7
		}
		change_variable = { name = government_leader_william_gladstone_counter add = 1 }
	}
	if = {
		limit = {
			has_variable = solve_british_empire_ireland
			ig:ig_intelligentsia ?= {
				is_in_government = yes
			}
			legitimacy >= 80
		}
		change_variable = { name = government_leader_william_gladstone_counter add = 1 }
	}
	if = {
		limit = {
			any_country = {
				AND = {
					capital = {
                   		is_in_geographic_region = geographic_region_europe
                	}
                	country_rank >= rank_value:great_power
                	is_at_war = yes
                	NOT = { country_has_primary_culture = cu:british }
				}
				count < 2
			}
		}
		change_variable = { name = government_leader_william_gladstone_counter add = 1 }
	}
	if = {
		limit = {
			OR = {
				has_law_or_variant = law_type:law_census_voting
				has_law_or_variant = law_type:law_universal_suffrage
			}
		}
		change_variable = { name = government_leader_william_gladstone_counter add = 1 }
	}
}
