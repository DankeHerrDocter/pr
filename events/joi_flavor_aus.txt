joi_flavor_aus.1 = {#进军布达佩斯
	type = country_event
	placement = root
	title = joi_flavor_aus.1.t
	desc = joi_flavor_aus.1.d
	flavor = joi_flavor_aus.1.f

	duration = 3

	dlc = dlc_vc

	event_image = {
		video = "ip3_officer_inspecting_peasant_conscripts"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	trigger = {
	}

	immediate = {
		hidden_effect = {
			add_modifier = { name = civil_war_clear_badname months = 1 }
		}
	}

	option = {

		name = joi_flavor_aus.1.a
		default_option = yes

		hidden_effect = {
			add_modifier = { name = civil_war_clear_badname months = 1 }
			every_scope_state = {
				limit = {
					has_modifier = hungarian_nobility_privileges
				}
				remove_modifier = hungarian_nobility_privileges				
			}
		}
		add_journal_entry = { type = marching_through_budapest }
		create_country = {
				tag = HUN
				origin = this
				state = s:STATE_CENTRAL_HUNGARY.region_state:AUS
				on_created = {
				activate_law = law_type:law_presidential_republic
				activate_law = law_type:law_national_supremacy
				activate_law = law_type:law_interventionism
				activate_law = law_type:law_landed_voting
				activate_law = law_type:law_local_police
				activate_law = law_type:law_mercantilism
				activate_law = law_type:law_no_social_security
				activate_law = law_type:law_no_home_affairs
				}
			}
		s:STATE_TRANSDANUBIA = {
			random_scope_state = {
				limit = {
					owner = ROOT
					}
				set_state_owner = c:HUN
				}
			}
		s:STATE_TRANSDANUBIA = {
			random_scope_state = {
				limit = {
					owner = ROOT
					}
				set_state_owner = c:HUN
				}
			}
		if = {
			limit = {
				var:hungarian_nobles_problem_var >= 25
			}
			s:STATE_BEKES = {
			random_scope_state = {
				limit = {
					owner = ROOT
					}
				set_state_owner = c:HUN
				}
			}
		}
		if = {
			limit = {
				var:hungarian_nobles_problem_var >= 50
			}
			s:STATE_SLAVONIA = {
			random_scope_state = {
				limit = {
					owner = ROOT
					}
				set_state_owner = c:HUN
				}
			}
			s:STATE_DELVIDEK = {
			random_scope_state = {
				limit = {
					owner = ROOT
					}
				set_state_owner = c:HUN
				}
			}
			s:STATE_BANAT = {
			random_scope_state = {
				limit = {
					owner = ROOT
					}
				set_state_owner = c:HUN
				}
			}
		}
		if = {
			limit = {
				var:hungarian_nobles_problem_var >= 75
			}
			s:STATE_NORTHERN_TRANSYLVANIA = {
			random_scope_state = {
				limit = {
					owner = ROOT
					}
				set_state_owner = c:HUN
				}
			}
			s:STATE_SOUTHERN_TRANSYLVANIA = {
			random_scope_state = {
				limit = {
					owner = ROOT
					}
				set_state_owner = c:HUN
				}
			}
		}
		c:AUS = {
		create_diplomatic_play = {
			name = aus_civil_war
		
			target_state = s:STATE_CENTRAL_HUNGARY.region_state:HUN
			
			war = yes
			
			type = dp_annex_war

			add_war_goal = {
				holder = c:HUN
				type = return_state 
				target_state = s:STATE_SLOVAKIA.region_state:AUS
					}
			add_war_goal = {
				holder = c:HUN
				type = return_state 
				target_state = s:STATE_RUTHENIA.region_state:AUS
				}
			}			
		}
		change_variable = { name = hungarian_nobles_problem_var add = -120 }
		clamp_variable = { name = hungarian_nobles_problem_var max = 120 min = 0 }
	}
}

joi_flavor_aus.25 = {#旧时代的帝国封地
	type = country_event
	placement = root
	title = joi_flavor_aus.25.t
	desc = joi_flavor_aus.25.d
	flavor = joi_flavor_aus.25.f

	duration = 3

	dlc = dlc_vc

	event_image = {
		video = "ip4_colonial_exploitation_going_well"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/joi_icons/house_of_habsburg.png"

	trigger = {
	}

	immediate = {
		
	}

	option = {

		trigger = {
			owns_entire_state_region = STATE_FLANDERS
			owns_entire_state_region = STATE_WALLONIA
		}

		name = joi_flavor_aus.25.a

		add_modifier = { name = recover_bel_modifier years = 10 }
		every_scope_state = {
			limit = {
				OR = {
					state_region = s:STATE_FLANDERS
					state_region = s:STATE_WALLONIA
				}
			}
			add_modifier = imperial_council_of_state_1_modifier
		}
		s:STATE_HOLLAND = {
			add_claim = ROOT
		}
		s:STATE_GELRE = {
			add_claim = ROOT
		}
		s:STATE_FRIESLAND = {
			add_claim = ROOT
		}
	}

	option = {

		name = joi_flavor_aus.25.b
		default_option = yes

		add_modifier = { name = recover_bel_modifier years = 10 }
		if = {
			limit = {
				owns_entire_state_region = STATE_FLANDERS
				NOT = { exists = c:BEL }
				c:AUS ?= THIS
				}
				create_country = {
				tag = BEL
				origin = this
				state = s:STATE_FLANDERS.region_state:AUS
				on_created = {
				activate_law = law_type:law_monarchy
				}
			}
		}
		if = {
			limit = {
				owns_entire_state_region = STATE_FLANDERS
				NOT = { exists = c:BEL }
				c:GER ?= THIS
				}
				create_country = {
				tag = BEL
				origin = this
				state = s:STATE_FLANDERS.region_state:GER
				on_created = {
				activate_law = law_type:law_monarchy
				}
			}
		}
		if = {
			limit = {
				owns_entire_state_region = STATE_FLANDERS
				NOT = { exists = c:BEL }
				c:HRE ?= THIS
				}
				create_country = {
				tag = BEL
				origin = this
				state = s:STATE_FLANDERS.region_state:HRE
				on_created = {
				activate_law = law_type:law_monarchy
				}
			}
		}
		if = {
			limit = { exists = c:BEL }
			every_scope_state = {
				limit = {
					OR = {
						state_region = s:STATE_WALLONIA
						state_region = s:STATE_FLANDERS
					}
				}
				set_state_owner = c:BEL
				set_state_type = incorporated
			}
		}
		if = {
			limit = { exists = c:BEL }
			create_diplomatic_pact = {
				country = c:BEL
				type = crown_land
			}
		}
		hidden_effect = {
			c:BEL ?= {
				add_modifier = aus_crown_land_modifier
			}
		}
		custom_tooltip = {
		text = event_aus_crown_land_tt
		}
		show_as_tooltip = {
			add_modifier = aus_crown_land_modifier
		}
		s:STATE_HOLLAND = {
			add_claim = ROOT
		}
		s:STATE_GELRE = {
			add_claim = ROOT
		}
		s:STATE_FRIESLAND = {
			add_claim = ROOT
		}
	}
}
