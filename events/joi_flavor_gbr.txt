	flavor = joi_flavor_gbr.1.f

	duration = 1

	dlc = dlc_vc

	event_image = {
		video = "unspecific_world_fair"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	trigger = {
	}

	immediate = {
		set_variable = {
			name = stabilize_british_empire_var
			value = 0
		}
	}

	option = {

		name = joi_flavor_gbr.1.a
		default_option = yes

		show_as_tooltip = {
			add_modifier = rule_britannia
		}
		hidden_effect = {
			set_global_variable = victorian_century_start_var
			set_new_literacy_rate_ire = yes
			c:CHI ?= {
				trigger_event = { id = opium_wars.1 }
			}
			every_country = {
				limit = {
					is_subject_of = ROOT
					OR = {
						has_law_or_variant = law_type:law_monarchy
						has_law_or_variant = law_type:law_social_monarchy
					}
					NOT = { c:BIC ?= THIS }
				}
				active_law:lawgroup_governance_principles ?= { # 威斯敏斯特体系
					add_amendment = {
						type = amendment_westminster_system_amendment
						sponsor = PREV.ig:ig_intelligentsia
					}
				}
				active_law:lawgroup_economic_system ?= { # 普通法
					add_amendment = {
						type = amendment_common_law
						sponsor = PREV.ig:ig_rural_folk
					}
				}
			}
		}
	}
}
