joi_flavor_tur.4 = {#欧洲死尸
	type = country_event
	placement = root
	title = joi_flavor_tur.4.t
	desc = joi_flavor_tur.4.d
	flavor = joi_flavor_tur.4.f

	duration = 3

	dlc = dlc_vc

	event_image = {
		video = "ip3_angry_mob_raiding_minorities"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	trigger = {
	}

	immediate = {
		set_variable = sick_man_complete
		remove_modifier = outmoded_bureaucracy
		ruler = {
			save_scope_as = ottoman_sultan
		}
	}

	option = {

		name = joi_flavor_tur.4.a
		default_option = yes
		
		ig:ig_intelligentsia = {
			remove_ideology = ideology_tanzimatism
		}
		ig:ig_armed_forces = {
			remove_ideology = ideology_tanzimatism
		}
		if = {
			limit = {
				is_subject = no
				any_subject_or_below = {
					NOT = { any_overlord_or_above = { NOT = { ROOT = this } } }
					NOT = {
						is_country_type = unrecognized
					}
				}
			}
			every_subject_or_below = {
				limit = {
					NOT = { any_overlord_or_above = { NOT = { ROOT = this } } }
					OR = {
						NOT = {
							is_country_type = unrecognized
						}
						is_subject_type = subject_type_protectorate
						is_subject_type = subject_type_dominion
						is_subject_type = subject_type_puppet
						is_subject_type = subject_type_colony
						is_subject_type = subject_type_chartered_company
					}
				}
				make_independent = yes
			}
		}
		set_country_type = unrecognized
		ruler = {
			kill_character = yes
		}
		if = { #failure of JE immediately triggers eastern_crisis, showing here
			limit = {
				has_dlc_feature = ip3_content
			}
			show_as_tooltip = {
				add_journal_entry = {
					type = je_great_eastern_crisis
				}
			}
		}
		custom_tooltip = {
			text = je_sick_man_main_failure_movements_tt
			set_variable = {
				name = failed_tanzimat
				years = 20
			}
		}
		if = {
			limit = {
				has_dlc_feature = ip3_content
			}
			custom_tooltip = {
				text = eastern_crisis_consequence_tt
				set_variable = failed_sick_man_main
			}
		}
		if = { #failure immediately triggers eastern_crisis
			limit = {
				c:TUR ?= this
				has_dlc_feature = ip3_content
				NOT = {
					has_variable = eastern_crisis_happened
				}
			}
			add_journal_entry = {
				type = je_great_eastern_crisis
			}
		}
		cu:albanian = {
			remove_tradition_trait = tradition_rumelian
		}
		cu:bosniak = {
			remove_tradition_trait = tradition_rumelian
		}
		cu:sephardic = {
			remove_tradition_trait = tradition_rumelian
		}
	}
}
